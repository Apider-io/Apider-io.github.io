my_automation.py
from apider import Email, Telegram, Sheets, CloudScheduler

def daily_report():
    emails = Email.read(unread_only=True)
    Sheets.append([e.subject for e in emails])
    Telegram.send(f"📬 {len(emails)} new emails processed")

# Runs on Azure — no server needed
CloudScheduler.daily_at("08:00", daily_report)
