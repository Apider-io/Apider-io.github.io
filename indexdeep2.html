from apider import Email

# Read unread emails
emails = Email.read(unread_only=True)
for email in emails:
    print(email.subject, email.sender)

# Send an email
Email.send(
    to="team@company.com",
    subject="Daily report",
    body="Attached is the report."
)
from apider import Telegram

# Send instant notification
Telegram.send("🚨 Alert: Server CPU > 80%")

# Poll messages from a group
messages = Telegram.poll(limit=10)
for msg in messages:
    print(msg.text)
from apider import Sheets

# Read data as DataFrame (Polars)
df = Sheets.read(sheet="Sales")
print(df.filter(df["revenue"] > 1000))

# Append a row
Sheets.append(["2025-03-01", "Product X", 2499])
