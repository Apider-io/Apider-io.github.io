from apider import Email, Telegram, Sheets, CloudScheduler

def daily_report():
    emails = Email.read(unread_only=True)
    Sheets.append([e.subject for e in emails])
    Telegram.send(f"📬 {len(emails)} new emails")

# Runs on Azure Functions — zero infrastructure
CloudScheduler.daily_at("09:00", daily_report)
