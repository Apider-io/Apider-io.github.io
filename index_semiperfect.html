from apider import Email

# Read latest unread emails
emails = Email.read(unread_only=True)
for email in emails:
    print(f"From: {email.sender} - {email.subject}")

# Send a notification
Email.send(
    to="team@apider.dev",
    subject="Daily digest",
    body=f"You have {len(emails)} new messages."
)
from apider import Telegram

# Send an instant alert
Telegram.send("🚨 Server alert: CPU usage 95%")

# Poll messages from a group
messages = Telegram.poll(limit=5)
for msg in messages:
    print(f"@{msg.from_user}: {msg.text}")
from apider import Sheets

# Read data as Polars DataFrame
df = Sheets.read(sheet="Sales Q1")
high_value = df.filter(df["revenue"] > 5000)

# Append a new row
Sheets.append(["2025-04-01", "Product Z", 12999])
